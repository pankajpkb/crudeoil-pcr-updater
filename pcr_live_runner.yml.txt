name: Live PCR Script Runner

on:
  push:
    branches:
      - main
  workflow_dispatch:  # manual trigger bhi possible
  schedule:
    - cron: "*/5 * * * *"  # har 5 minute run, optional

jobs:
  run-pcr:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout Repository
        uses: actions/checkout@v3

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: 3.10

      - name: Upgrade pip
        run: python -m pip install --upgrade pip

      - name: Install Python Dependencies
        run: |
          pip install requests beautifulsoup4 pandas gspread oauth2client

      - name: Create credentials.json from Secret
        run: |
          echo "${{ secrets.GOOGLE_CREDENTIALS }}" | sed 's/\\n/\n/g' > credentials.json

      - name: Run PCR Script
        run: python Pcr_File_Run_On_Cloud.py
